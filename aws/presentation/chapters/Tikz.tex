\newcommand{\bucket}[4]{
	\coordinate (C) at (#1,#2);
	\draw (C) ellipse (1.2*#3 and 0.5*#3);
	\coordinate (BL) at ($(C) + (-1*#3,-2*#3)$);
	\coordinate (BR) at ($(C) + (1*#3,-2*#3)$);
	\draw (BL) .. controls ($(C) - (0.4*#3,2.3*#3)$) and ($(C) - (-0.4*#3,2.3*#3)$) .. (BR);
	\coordinate (TL) at ($(C) + (-1.2*#3,0)$);
	\coordinate (TR) at ($(C) + (1.2*#3,0)$);
	\draw (BR) -- (TR);
	\draw (BL) -- (TL);
	\node (C) at (#1,#2-3*#3) {#4};
}

\newcommand{\file}[4]{
	\coordinate (C) at (#1,#2);
	\coordinate (BL) at ($(C) + (-1*#3,-1.5*#3)$);
	\coordinate (BR) at ($(C) + (1*#3,-1.5*#3)$);
	\coordinate (TL) at ($(C) + (-1*#3,1.5*#3)$);
	\coordinate (TR1) at ($(C) + (0.3*#3,1.5*#3)$);
	\coordinate (TR2) at ($(C) + (1*#3,0.8*#3)$);
	\coordinate (TR3) at ($(C) + (0.3*#3,0.8*#3)$);
	\draw (TR1) -- (TL) -- (BL) -- (BR) -- (TR2) -- (TR3) -- (TR1) -- (TR2);
	\draw ($(C) + (-0.8*#3,0.5*#3)$) -- ($(C) + (0.8*#3,0.5*#3)$);
	\draw ($(C) + (-0.8*#3,0.2*#3)$) -- ($(C) + (0.8*#3,0.2*#3)$);
	\draw ($(C) + (-0.8*#3,-0.1*#3)$) -- ($(C) + (0.8*#3,-0.1*#3)$);
	\draw ($(C) + (-0.8*#3,-0.4*#3)$) -- ($(C) + (0.8*#3,-0.4*#3)$);
	\draw ($(C) + (-0.8*#3,-0.7*#3)$) -- ($(C) + (0.8*#3,-0.7*#3)$);
	\node at ($(C) - (0,2*#3)$) {#4};
}
